Prototypes vs. Classes


var Mammal = function() {};
Mammal.prototype.neoCortex = true;

var Cat = function(name, years) {
  this.name = name;
  this.years = years;
};

Cat.prototype = new Mammal;

var fryCat = new Cat('Fry', 7);

fryCat.prototype.eat = function(food) {
  console.log('nom ' + food);
}
fryCat.eat('steak');
								
class Mammal(object):
neo_cortex = True

class Cat(Mammal):
  def __init__(self, name, years):
    self.name = name
    self.years = years

def eat(food):
  print 'nom %s' % (food)

fry_cat = Cat('Fry', 7)

fry_cat.eat('steak')
