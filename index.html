Ternary


var Mammal = function() {};
Mammal.prototype.neoCortex = true;

var Cat = function(name, years) {
  this.name = name;
  this.years = years;
};

Cat.prototype = new Mammal;

var fryCat = new Cat('Fry', 7);

fryCat.prototype.eat = function(food) {
    console.log('nom nom nom ' + food);
}

fryCat.eat('steak');

						

class Mammal(object):
    neo_cortex = True

class Cat(Mammal):
    def __init__(self, name, years):
      self.name = name
      self.years = years

    def eat(food):
        print 'nom nom nom %s' % (food)

fry_cat = Cat('Fry', 7)

fry_cat.eat('steak')

Scope

// JavaScript uses Lexical Scope // JavaScript only creates scope (namespace) with function definitions // New identifiers affected by scope are variables and functions. // New JavaScript files do not get their own scope!

# Python uses Lexical Scope # Python creates scope (namespace) using functions, classes, and modules. # New identifiers affected by scope are variables, functions, and classes. # All new Python files are Modules, thus receiving their own scope.
